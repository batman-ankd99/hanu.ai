SELECT
   id,
   effect,
   principal, actions, conditions, statement_id,
   policy_arn

FROM iam_policy_statements
WHERE is_action_star='t' OR is_principal_star='t';


JOIN

SELECT
    s.id,
    s.effect,
    s.principal,
    s.actions,
    p.policy_name,
    p.attached_entities
FROM iam_policy_statements AS s
JOIN iam_policies AS p
    ON s.policy_arn = p.policy_arn
WHERE (s.is_action_star = 't' OR s.is_principal_star = 't')
  AND s.effect = 'Allow';
